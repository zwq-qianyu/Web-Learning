x = ['q1', 'q2', 'q3', 'q4']
y = [10, 30, 20, 60]
rect = plt.bar(x, y, color='c', width=0.3)
for ind, item in enumerate(rect):
    _x = item.get_x() + 0.1
    _y = item.get_height() + 2
    plt.text(_x, _y, y[ind])
plt.ylim(0, 70)     # y轴取值范围为0到70
plt.show()
